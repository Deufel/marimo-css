    funccli.lint_notebook

    Extract CSS from a marimo notebook and lint it.

    def lint_notebook(notebook_path: str) -> Report:    """Extract CSS from a marimo notebook and lint it."""    css = get_css(notebook_path)    # write to temp file so lint_file can reference it    tmp = Path(notebook_path).with_suffix(".css")    tmp.write_text(css)    report = Report()    lint_file(tmp, report, tmp.parent)    tmp.unlink()    return report
