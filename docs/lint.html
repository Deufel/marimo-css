    funclint.find_root
    def find_root(start: Path = None, marker: str = "pyproject.toml") -> Path:    p = (start or Path.cwd()).resolve()    for parent in [p, *p.parents]:        if (parent / marker).exists():            return parent    raise FileNotFoundError(f"No {marker} above {p}")
